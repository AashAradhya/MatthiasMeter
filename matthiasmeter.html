function updateTeamDashboard() { const allCharges = JSON.parse(localStorage.getItem('charges') || '{}'); const users = Object.keys(allCharges); if (users.length === 0) { document.getElementById('teamSummary').innerHTML = '
No team members have added charges yet.
'; document.getElementById('teamBreakdown').innerHTML = ''; return; } // Calculate team totals let teamTotalUSD = 0; let teamTotalGold = 0; const userTotals = {}; users.forEach(user => { const userCharges = allCharges[user] || []; userTotals[user] = { totalUSD: 0, totalGold: 0, byCategory: {}, chargeCount: userCharges.length }; userCharges.forEach(charge => { if (charge.units === 'USD') { userTotals[user].totalUSD += charge.amount; teamTotalUSD += charge.amount; } else if (charge.units === 'SPDR Gold') { userTotals[user].totalGold += charge.amount; teamTotalGold += charge.amount; } if (!userTotals[user].byCategory[charge.category]) { userTotals[user].byCategory[charge.category] = { count: 0, amount: 0, units: charge.units }; } userTotals[user].byCategory[charge.category].count++; userTotals[user].byCategory[charge.category].amount += charge.amount; }); }); // Render team summary let summaryHTML = `

Total Team USD

$${teamTotalUSD.toFixed(2)}
${users.length} team members

Total SPDR Gold

${teamTotalGold.toFixed(2)}
across all members
`; document.getElementById('teamSummary').innerHTML = summaryHTML; // Render breakdown by user let breakdownHTML = ''; Object.keys(userTotals).sort((a, b) => { const aTotal = userTotals[a].totalUSD + (userTotals[a].totalGold * 100); // Rough gold valuation for sorting const bTotal = userTotals[b].totalUSD + (userTotals[b].totalGold * 100); return bTotal - aTotal; }).forEach(user => { const data = userTotals[user]; breakdownHTML += `

${user}

USD: $${data.totalUSD.toFixed(2)}

Gold: ${data.totalGold.toFixed(2)}

Charges: ${data.chargeCount}

${Object.keys(data.byCategory).map(cat => { const catData = data.byCategory[cat]; return ` ${cat}: ${catData.units === 'USD' ? '$' + catData.amount.toFixed(2) : catData.amount + ' ' + catData.units} (${catData.count}x) `; }).join('')}
`; }); document.getElementById('teamBreakdown').innerHTML = breakdownHTML; }
