To make MatthiasMeter work for your team, follow these steps:

  1. Go to Firebase Console
  2. Create a new project (or use existing one)
  3. Enable Realtime Database in the Firebase console
  4. Set Database Rules to: 
    {
  "rules": {
    ".read": true,
    ".write": true
  }
}
  5. Go to Project Settings â†’ Add Web App
  6. Copy your Firebase config and paste it in the JavaScript section of this HTML file (search for "PASTE YOUR FIREBASE CONFIG HERE")
  7. Save and refresh the page

// ============================================ // Initialize Firebase let db = null; let currentUser = null; let charges = []; let allUsersData = {}; function initializeApp() { // Check if config is set if (firebaseConfig.apiKey === "YOUR_API_KEY") { document.getElementById('loadingView').classList.add('hidden'); document.getElementById('setupView').classList.remove('hidden'); return; } try { firebase.initializeApp(firebaseConfig); db = firebase.database(); // Listen for data changes db.ref('users').on('value', (snapshot) => { const users = snapshot.val() || {}; updateUserList(Object.keys(users)); }); db.ref('charges').on('value', (snapshot) => { allUsersData = snapshot.val() || {}; if (currentUser) { charges = allUsersData[currentUser] || []; updateDashboard(); } }); document.getElementById('loadingView').classList.add('hidden'); // Check for saved user const savedUser = localStorage.getItem('currentUser'); if (savedUser) { currentUser = savedUser; showDashboard(); } else { document.getElementById('authView').classList.remove('hidden'); } } catch (error) { console.error('Firebase initialization error:', error); document.getElementById('loadingView').innerHTML = `

Firebase Error

${error.message}

Please check your Firebase configuration.

`; } } function updateUserList(users) { const select = document.getElementById('userSelect'); const currentValue = select.value; select.innerHTML = ''; users.forEach(user => { const option = document.createElement('option'); option.value = user; option.textContent = user; select.appendChild(option); }); if (currentValue) { select.value = currentValue; } } function updateLoginButton() { const select = document.getElementById('userSelect'); document.getElementById('selectUserBtn').disabled = !select.value; } function selectUser() { const select = document.getElementById('userSelect'); const username = select.value; if (!username) { alert('Please select your name'); return; } currentUser = username; localStorage.setItem('currentUser', username); showDashboard(); } async function addNewUser() { const newName = document.getElementById('newUserName').value.trim(); if (!newName) { alert('Please enter your name'); return; } try { // Check if user exists const snapshot = await db.ref('users/' + newName).once('value'); if (snapshot.exists()) { alert('This name already exists. Please select it from the dropdown.'); return; } // Add new user await db.ref('users/' + newName).set({ createdAt: new Date().toISOString() }); // Initialize empty charges await db.ref('charges/' + newName).set([]); currentUser = newName; localStorage.setItem('currentUser', newName); charges = []; showDashboard(); } catch (error) { console.error('Error adding user:', error); alert('Failed to add user. Please try again.'); } } function showPersonalView() { document.getElementById('personalView').classList.remove('hidden'); document.getElementById('teamView').classList.add('hidden'); document.getElementById('personalBtn').style.background = 'linear-gradient(135deg, #667eea 0%, #764ba2 100%)'; document.getElementById('teamBtn').style.background = '#6c757d'; } function showTeamView() { document.getElementById('personalView').classList.add('hidden'); document.getElementById('teamView').classList.remove('hidden'); document.getElementById('personalBtn').style.background = '#6c757d'; document.getElementById('teamBtn').style.background = 'linear-gradient(135deg, #667eea 0%, #764ba2 100%)'; updateTeamDashboard(); } function logout() { currentUser = null; localStorage.removeItem('currentUser'); document.getElementById('authView').classList.remove('hidden'); document.getElementById('dashboardView').classList.add('hidden'); } function showDashboard() { document.getElementById('authView').classList.add('hidden'); document.getElementById('dashboardView').classList.remove('hidden'); document.getElementById('currentUser').textContent = currentUser; showPersonalView(); updateDashboard(); } async function quickAddCharge(category) { let amount = 0; let units = 'USD'; let delayHours = null; let minutesLate = null; let defaultDesc = ''; if (category === 'Missed Message') { delayHours = parseFloat(document.getElementById('quickDelay').value) || 2; amount = (delayHours / 2) * 10; defaultDesc = `Message delayed by ${delayHours} hours`; } else if (category === 'Late to Meeting') { minutesLate = parseInt(document.getElementById('quickMinutes').value) || 5; amount = minutesLate * 2; defaultDesc = `Joined meeting ${minutesLate} minutes late`; } else if (category === 'Failed Notebook Action') { amount = 1; units = 'SPDR Gold'; defaultDesc = 'Notebook action not completed'; } if (amount <= 0) { alert('Please enter valid delay/late time'); return; } const charge = { id: Date.now(), category, description: defaultDesc, amount, units, delayHours, minutesLate, createdAt: new Date().toISOString() }; try { charges.unshift(charge); await db.ref('charges/' + currentUser).set(charges); // Clear inputs if (category === 'Missed Message') { document.getElementById('quickDelay').value = ''; } else if (category === 'Late to Meeting') { document.getElementById('quickMinutes').value = ''; } // Show success message const successEl = document.getElementById('addChargeSuccess'); successEl.textContent = `Added ${category}! ${units === 'USD' ? '$' + amount.toFixed(2) : amount + ' ' + units}`; successEl.classList.remove('hidden'); setTimeout(() => successEl.classList.add('hidden'), 2000); } catch (error) { console.error('Error adding charge:', error); alert('Failed to add charge. Please try again.'); } } async function deleteCharge(chargeId) { if (!confirm('Are you sure you want to delete this charge?')) { return; } try { charges = charges.filter(c => c.id !== chargeId); await db.ref('charges/' + currentUser).set(charges); } catch (error) { console.error('Error deleting charge:', error); alert('Failed to delete charge. Please try again.'); } } function updateDashboard() { // Update summary let totalUSD = 0; let totalGold = 0; const summary = {}; charges.forEach(charge => { const key = `${charge.category}|${charge.units}`; if (!summary[key]) { summary[key] = { category: charge.category, units: charge.units, total: 0, count: 0 }; } summary[key].total += charge.amount; summary[key].count++; if (charge.units === 'USD') { totalUSD += charge.amount; } else if (charge.units === 'SPDR Gold') { totalGold += charge.amount; } }); // Render summary let summaryHTML = `

Total in USD

$${totalUSD.toFixed(2)}
${charges.filter(c => c.units === 'USD').length} charges

SPDR Gold Trust

${totalGold.toFixed(2)}
${charges.filter(c => c.units === 'SPDR Gold').length} actions
`; Object.values(summary).forEach(item => { summaryHTML += `

${item.category}

${item.units === 'USD' ? '$' + item.total.toFixed(2) : item.total.toFixed(2) + ' ' + item.units}
${item.count} entries
`; }); document.getElementById('summaryGrid').innerHTML = summaryHTML; // Update charges list document.getElementById('chargeCount').textContent = charges.length; if (charges.length === 0) { document.getElementById('chargesList').innerHTML = '
No charges yet. Start by adding your first charge above!
'; } else { let chargesHTML = ''; charges.forEach((charge) => { const date = new Date(charge.createdAt); const dateStr = date.toLocaleDateString() + ' ' + date.toLocaleTimeString(); chargesHTML += `

${charge.category}

${charge.description}

${charge.delayHours ? `

Delay: ${charge.delayHours} hours

` : ''} ${charge.minutesLate ? `

Minutes late: ${charge.minutesLate}

` : ''}
${charge.units === 'USD' ? '$' + charge.amount.toFixed(2) : charge.amount + ' ' + charge.units}
${dateStr}
`; }); document.getElementById('chargesList').innerHTML = chargesHTML; } } function updateTeamDashboard() { const users = Object.keys(allUsersData); if (users.length === 0) { document.getElementById('teamSummary').innerHTML = '
No team members have added charges yet.
'; document.getElementById('teamBreakdown').innerHTML = ''; return; } // Calculate team totals let teamTotalUSD = 0; let teamTotalGold = 0; const userTotals = {}; users.forEach(user => { const userCharges = allUsersData[user] || []; userTotals[user] = { totalUSD: 0, totalGold: 0, byCategory: {}, chargeCount: userCharges.length }; userCharges.forEach(charge => { if (charge.units === 'USD') { userTotals[user].totalUSD += charge.amount; teamTotalUSD += charge.amount; } else if (charge.units === 'SPDR Gold') { userTotals[user].totalGold += charge.amount; teamTotalGold += charge.amount; } if (!userTotals[user].byCategory[charge.category]) { userTotals[user].byCategory[charge.category] = { count: 0, amount: 0, units: charge.units }; } userTotals[user].byCategory[charge.category].count++; userTotals[user].byCategory[charge.category].amount += charge.amount; }); }); // Render team summary let summaryHTML = `

Total Team USD

$${teamTotalUSD.toFixed(2)}
${users.length} team members

Total SPDR Gold

${teamTotalGold.toFixed(2)}
across all members
`; document.getElementById('teamSummary').innerHTML = summaryHTML; // Render breakdown by user let breakdownHTML = ''; Object.keys(userTotals).sort((a, b) => { const aTotal = userTotals[a].totalUSD + (userTotals[a].totalGold * 100); const bTotal = userTotals[b].totalUSD + (userTotals[b].totalGold * 100); return bTotal - aTotal; }).forEach(user => { const data = userTotals[user]; breakdownHTML += `

${user}

USD: $${data.totalUSD.toFixed(2)}

Gold: ${data.totalGold.toFixed(2)}

Charges: ${data.chargeCount}

${Object.keys(data.byCategory).map(cat => { const catData = data.byCategory[cat]; return ` ${cat}: ${catData.units === 'USD' ? '$' + catData.amount.toFixed(2) : catData.amount + ' ' + catData.units} (${catData.count}x) `; }).join('')}
`; }); document.getElementById('teamBreakdown').innerHTML = breakdownHTML; } // Initialize app on load window.onload = initializeApp;
